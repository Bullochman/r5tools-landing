LWSLineChart — SVG line chart

Zero-dependency vanilla-JS SVG line chart used across r5tools.io for roster power trends and alliance leaderboards. Under 260 lines, retina-crisp, dark-theme native.

Alliance Power Trend Single line · 8 snapshots · auto-scaled Y axis, rotated X labels, hover tooltip

Top 5 Alliances in Warzone 2007 Multi-line with legend · color-coded series · shared axes

Empty state Graceful fallback when no data is available

How to embed

<!-- 1. drop in the module (no build step) -->
<script src="/line-chart.js"></script>

<!-- 2. mount -->
<div id="my-chart"></div>

<script>
// single-line
LWSLineChart.render(document.getElementById('my-chart'), {
  title: 'Alliance Power',
  labelX: 'Snapshot date',
  labelY: 'Power',
  series: [
    { x: '2026-07-01', y: 812000000 },
    { x: '2026-07-08', y: 894000000 },
    { x: '2026-07-15', y: 1020000000 }
  ]
});

// multi-line with legend
LWSLineChart.renderMulti(document.getElementById('my-chart'), {
  title: 'Top alliances',
  series: [
    { label: 'RONY', color: '#c9a961', points: [{x:1,y:100},{x:2,y:140}] },
    { label: 'HIVE', color: '#8ae0a3', points: [{x:1,y:80},{x:2,y:110}] }
  ]
});
</script>
